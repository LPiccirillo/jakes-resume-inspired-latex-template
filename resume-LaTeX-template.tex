% DOCUMENT CLASS
% Defines the base document type and global page settings.
% extarticle is an extended version of the standard LaTeX
% article class, allowing additional font sizes.
% Arguments:
%   a4paper -> uses A4 page dimensions.
%   9pt     -> sets the base font size to 9 points, useful
%              for compact CV layouts.
\documentclass[a4paper,9pt]{extarticle}

% CUSTOM CV STYLE PACKAGE
% Loads the custom style file containing all formatting
% definitions and reusable commands for the CV.
% The resume_style.sty file includes:
% - required LaTeX packages;
% - page layout and margin settings;
% - colours and typography settings;
% - header/footer configuration;
% - section formatting;
% - custom commands for CV entries, bullet points
%   and structured sections.
% Keeping these definitions in a separate .sty file makes
% the main .tex document cleaner and allows the same CV
% template to be reused for different documents.
\usepackage{resume-style}
